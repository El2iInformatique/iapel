12/**
13 * @class VerifTokenMiddleware
14 * @brief Middleware de vérification des tokens d'accès aux rapports et documents.
15 *
16 * Ce middleware assure la sécurité d'accès aux documents sensibles (rapports d'intervention,
17 * formulaires CERFA) en validant les tokens fournis dans les URLs. Il constitue la première
18 * couche de protection pour empêcher l'accès non autorisé aux documents générés par l'application.
19 *
20 * Fonctionnalités principales :
21 * - Validation de l'existence et de la validité des tokens dans les routes.
22 * - Vérification de l'expiration des tokens d'accès.
23 * - Logging automatique des tentatives d'accès pour audit de sécurité.
24 * - Blocage immédiat des accès avec tokens invalides ou expirés.
25 * - Intégration transparente avec le système de routage Laravel.
26 * - Support des tokens de type TokenLinksRapport pour documents BI.
27 *
28 * @package App\Http\Middleware
29 * @version 1.0
30 * @author Maxime ENTZ
31 * @since 1.0
32 * @note Utilisé pour protéger les routes d'accès aux documents via tokens.
33 * @note Logs toutes les tentatives de vérification pour traçabilité.
34 * @warning Retourne une erreur 404 si le token est invalide pour masquer l'existence de la ressource.
35 * @see TokenController::isValideTokenRapport() Pour la logique de validation des tokens.
36 * @see TokenLinksRapport Pour le modèle de gestion des tokens de rapports.
37 * @example
38 * ```php
39 * // Utilisation dans les routes (routes/web.php)
40 * Route::get('/bi/{token}', [BiController::class, 'show'])
41 * ->middleware('verify.token');
42 *
43 * Route::get('/download/{token}', [BiController::class, 'download'])
44 * ->middleware('verify.token');
45 * ```
46 */

49 /**
50 * @brief Traite une requête entrante et vérifie la validité du token.
51 *
52 * Cette méthode constitue le point d'entrée du middleware. Elle extrait le token
53 * depuis les paramètres de route, le valide via TokenController, et décide
54 * d'autoriser ou de bloquer l'accès à la ressource demandée.
55 *
56 * Le processus de vérification suit ces étapes :
57 * 1. Extraction du token depuis les paramètres de route.
58 * 2. Logging de la tentative d'accès pour audit.
59 * 3. Validation du token via TokenController::isValideTokenRapport().
60 * 4. Autorisation ou refus d'accès selon le résultat de validation.
61 *
62 * @param Request $request Requête HTTP entrante contenant le token en paramètre de route.
63 * @param Closure $next Fonction de callback pour passer au middleware suivant ou au contrôleur.
64 *
65 * @return Response Réponse HTTP (404 si token invalide, sinon passage au contrôleur).
66 *
67 * @throws \Symfony\Component\HttpKernel\Exception\NotFoundHttpException Si le token est invalide.
68 *
69 * @example
70 * ```php
71 * // URL d'accès typique
72 * https://app.example.com/bi/abc123def456...
73 *
74 * // Le middleware extrait 'abc123def456...' et le valide
75 * // Si valide : accès autorisé au BiController::show()
76 * // Si invalide : erreur 404 retournée
77 * ```
78 *
79 * @example
80 * ```php
81 * // Configuration dans le kernel HTTP (app/Http/Kernel.php)
82 * protected $middlewareAliases = [
83 * 'verify.token' => \App\Http\Middleware\VerifTokenMiddleware::class,
84 * ];
85 *
86 * // Protection de multiple routes
87 * Route::group(['middleware' => 'verify.token'], function () {
88 * Route::get('/bi/{token}', [BiController::class, 'show']);
89 * Route::get('/download/{token}', [BiController::class, 'download']);
90 * Route::post('/submit/{token}', [BiController::class, 'submit']);
91 * });
92 * ```
93 *
94 * @note Le token est automatiquement extrait depuis $request->route('token').
95 * @note Toutes les tentatives sont loggées avec le niveau INFO pour audit.
96 * @note L'erreur 404 est préférée à 403 pour ne pas révéler l'existence de la ressource.
97 * @warning Ne pas utiliser ce middleware sur des routes ne contenant pas de paramètre {token}.
98 * @warning Les logs peuvent contenir des tokens sensibles, attention à leur protection.
99 * @see Log::info() Pour l'enregistrement des tentatives d'accès.
100 * @see abort() Pour le déclenchement de l'erreur HTTP 404.
101 */
102 public function handle(Request $request, Closure $next): Response
103 {
104 $token = $request->route('token');
105 Log::info("Demande de verification du token : " . $token);
106
107 if (!TokenController::isValideTokenRapport($token)) {
108 abort(404, 'Token invalide');
109 }
110
111 return $next($request);
112 }
113}
