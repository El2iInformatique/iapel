8/**
9 * @class TokenLinksRapport
10 * @brief Modèle de gestion des tokens d'accès aux rapports et documents d'intervention.
11 *
12 * Ce modèle centralise la gestion des tokens sécurisés permettant l'accès aux rapports
13 * d'intervention, formulaires CERFA et autres documents générés par l'application.
14 * Il établit la liaison entre un token unique et le chemin physique du document associé.
15 *
16 * Fonctionnalités principales :
17 * - Génération de tokens d'accès sécurisés avec durée de validité.
18 * - Liaison token/chemin de fichier pour l'accès aux documents.
19 * - Gestion de l'expiration automatique des accès.
20 * - Support des rapports d'intervention, CERFA 15497, CERFA 13948-03.
21 * - Traçabilité des accès aux documents sensibles.
22 *
23 * @package App\Models
24 * @version 2.0
25 * @author Maxime ENTZ
26 * @since 1.0
27 * @note Les tokens expirent automatiquement après un mois pour des raisons de sécurité.
28 * @note La table utilisée est "token_links_rapport" et n'utilise pas les timestamps Laravel.
29 * @warning Les chemins stockés doivent être relatifs au répertoire storage/app/ de Laravel.
30 * @see BiController Pour l'utilisation des tokens dans le contexte des rapports.
31 * @see TokenController Pour la génération des tokens de signature de devis.
32 */

61 /**
62 * @brief Génère un nouveau token d'accès pour un rapport/document.
63 *
64 * Cette méthode statique crée un nouveau token d'accès sécurisé lié à un document
65 * spécifique. Le token permet l'accès au document via une URL sécurisée pendant
66 * une durée limitée d'un mois.
67 *
68 * Le token généré permet :
69 * - L'affichage du formulaire/rapport via une URL publique.
70 * - La soumission des données du formulaire.
71 * - Le téléchargement du PDF généré.
72 * - L'accès sécurisé sans authentification utilisateur.
73 *
74 * @param string $token Token unique généré (généralement par Str::random()).
75 * @param string $path Chemin relatif vers le fichier JSON du document (depuis storage/app/).
76 *
77 * @return TokenLinksRapport Instance du modèle créé avec les données du token.
78 *
79 * @throws \Illuminate\Database\QueryException Si la création en base échoue.
80 *
81 * @example
82 * ```php
83 * // Génération d'un token pour un rapport d'intervention
84 * $token = Str::random(40);
85 * $path = "app/public/client1/rapport_intervention/uid123/uid123.json";
86 *
87 * $tokenLink = TokenLinksRapport::generateTokenRapport($token, $path);
88 *
89 * // URL d'accès générée : /bi/{token}
90 * $url = url('/bi/' . $token);
91 * ```
92 *
93 * @example
94 * ```php
95 * // Génération d'un token pour un formulaire CERFA
96 * $token = Str::random(40);
97 * $path = "app/public/entreprise1/cerfa_15497/uid456/uid456.json";
98 *
99 * $tokenLink = TokenLinksRapport::generateTokenRapport($token, $path);
100 * ```
101 *
102 * @note Le token expire automatiquement après un mois pour des raisons de sécurité.
103 * @note La date de création est stockée au format Y-m-d.
104 * @warning Le chemin doit pointer vers un fichier JSON valide et existant.
105 * @see BiController::createJson() Méthode qui utilise cette fonction pour créer les tokens.
106 */
107 public static function generateTokenRapport( $token, $path ) {
108 return self::create([
109 'token' => $token,
110 'paths' => $path,
111 'expires_at' => now()->addMonth(),
112 'created_at' => now()->format('Y-m-d'),
113 ]);
114 }
