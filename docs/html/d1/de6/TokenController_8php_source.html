35 /**
36 * @brief Génère un token sécurisé pour l'accès à la signature de devis.
37 *
38 * Cette méthode crée un token d'accès unique avec toutes les informations nécessaires :
39 * - Authentification via secret token de l'organisation ou administrateur.
40 * - Validation complète des données d'entrée (montants, coordonnées, pagination).
41 * - Génération d'un token aléatoire avec date d'expiration automatique.
42 * - Stockage sécurisé des coordonnées de signature au format JSON.
43 * - Création automatique de l'URL de signature accessible au client.
44 *
45 * @param Request $request Requête POST contenant :
46 * - organisation_id : Identifiant de l'organisation (obligatoire)
47 * - devis_id : Identifiant unique du devis (obligatoire)
48 * - tiers : Nom du tiers/client (obligatoire)
49 * - client_email : Email du client (obligatoire)
50 * - titre : Titre/description du devis (obligatoire)
51 * - montant_HT : Montant hors taxes (numérique ≥ 0)
52 * - montant_TVA : Montant de la TVA (numérique ≥ 0)
53 * - montant_TTC : Montant toutes taxes comprises (numérique ≥ 0)
54 * - coords : Coordonnées de signature au format JSON
55 * - nb_pages : Nombre de pages du document (numérique ≥ 0)
56 *
57 * @return mixed Réponse JSON avec le token généré et l'URL de signature ou erreur.
58 *
59 * @throws Exception Si l'authentification échoue (403).
60 * @throws Exception Si les données JSON sont malformées (400).
61 * @throws Exception Si la validation des champs obligatoires échoue.
62 *
63 * @note L'en-tête 'secret-token' est obligatoire pour l'authentification.
64 * @note Les coordonnées doivent être un JSON valide contenant les positions de signature.
65 * @note Le token généré est automatiquement loggé pour audit de sécurité.
66 * @warning Les secret tokens doivent être configurés dans le fichier secrets.php.
67 * @par Exemple:
68 * POST avec secret-token header génère un token pour signature électronique.
69 */
70 public function generate(Request $request)
71 {
72 $request->validate([
73 'organisation_id' => 'required|string',
74 'devis_id' => 'required|string',
75 'tiers' => 'required|string',
76 'client_email' => 'required|string',
77 'titre' => 'required|string',
78 'montant_HT' => 'required|numeric|min:0',
79 'montant_TVA' => 'required|numeric|min:0',
80 'montant_TTC' => 'required|numeric|min:0',
81 'coords' => 'required|json',
82 'nb_pages' => 'required|numeric|min:0',
83 ]);
84
85 if (!$request->hasHeader('secret-token')) {
86 return response()->json(['error' => 'No secret token provided.'], 403);
87 }
88
89 $secretToken = config("secrets.$request->organisation_id");
90 $adminToken = config('secrets.admin');
91
92 $providedToken = $request->header('secret-token');
93
94 if (!hash_equals($providedToken, $secretToken) && !hash_equals($providedToken, $adminToken)) {
95 return response()->json(['error' => 'Not authorized.'], 403);
96 }
97
98 Log::info('Début de la génération du token', ['request_data' => $request->all()]);
99
100 $coords = json_decode($request->input('coords'), true);
101
102 if (json_last_error() !== JSON_ERROR_NONE) {
103 return response()->json(['error' => 'Invalid JSON in coords'], 400);
104 }
105
106 $token = Token::generateToken(
107 $request->organisation_id,
108 $request->tiers,
109 $request->devis_id,
110 $request->client_email,
111 $request->titre,
112 $request->montant_HT,
113 $request->montant_TVA,
114 $request->montant_TTC,
115 $coords,
116 $request->nb_pages,
117 );
118
119 return response()->json([
120 'message' => 'Token généré avec succès',
121 'token' => $token,
122 'signature_url' => url('/signature/' . $token->token),
123 ]);
124 }

126 /**
127 * @brief Valide un token d'accès avant l'affichage de l'interface de signature.
128 *
129 * Cette méthode de sécurité vérifie :
130 * - L'existence du token dans la base de données.
131 * - Le statut d'utilisation (non utilisé = false).
132 * - La date d'expiration (non expiré).
133 * - Retourne les informations du devis si valide.
134 *
135 * @param string $token Token unique à valider.
136 *
137 * @return mixed Réponse JSON avec validation et ID devis ou erreur 403.
138 *
139 * @throws Exception Si le token est invalide, utilisé ou expiré (403).
140 *
141 * @note Cette méthode est appelée avant l'affichage de l'interface de signature.
142 * @note Un token devient invalide dès qu'il est utilisé ou expire.
143 * @warning Les tokens expirés ne sont pas automatiquement supprimés ici.
144 * @par Exemple:
145 * GET /validate/ABC123 retourne la validité du token "ABC123".
146 */
147 public function validateToken($token)
148 {
149 $tokenEntry = Token::where('token', $token)->where('used', false)->where('expires_at', '>', now())->first();
150
151 if (!$tokenEntry) {
152 return response()->json(['message' => 'Token invalide ou expiré'], Response::HTTP_FORBIDDEN);
153 }
154
155 return response()->json(['message' => 'Token valide', 'devis_id' => $tokenEntry->devis_id]);
156 }

210 /**
211 * @brief Vérifie la validité d'un token de rapport et gère l'expiration automatique.
212 *
213 * Cette méthode statique de sécurité :
214 * - Recherche le token dans la table TokenLinksRapport.
215 * - Vérifie la date d'expiration avec comparaison précise.
216 * - Supprime automatiquement les tokens expirés de la base.
217 * - Journalise les tentatives d'accès avec tokens invalides.
218 *
219 * @param string $token Token de rapport à vérifier.
220 *
221 * @return bool True si le token est valide, False sinon.
222 *
223 * @note Les tokens expirés sont automatiquement supprimés lors de la vérification.
224 * @note L'expiration est vérifiée avec Carbon pour une précision maximale.
225 * @note Les accès avec tokens invalides sont loggés pour sécurité.
226 * @warning Un token supprimé pour expiration ne peut plus être récupéré.
227 * @par Exemple:
228 * isValideTokenRapport('ABC123') retourne true si le token existe et n'est pas expiré.
229 */
230 public static function isValideTokenRapport($token)
231 {
232 $tokenRecord = TokenLinksRapport::where('token', $token)->first();
233
234 if (!$tokenRecord) {
235 return false;
236 }
237
238 $expiresAt = Carbon::parse($tokenRecord->expires_at);
239
240 if ($expiresAt->lessThan(now())) {
241 Log::info('Token invalide : date depasser');
242
243 $tokenRecord->delete();
244 return false;
245 }
246
247 return true;
248 }
