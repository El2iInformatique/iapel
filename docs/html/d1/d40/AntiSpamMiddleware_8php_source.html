10/**
11 * @class AntiSpamMiddleware
12 * @brief Middleware de protection contre les attaques par déni de service et le spam.
13 *
14 * Ce middleware implémente un système de limitation de taux (rate limiting) pour protéger
15 * l'application contre les tentatives de spam et les attaques par déni de service (DoS).
16 * Il utilise le système de cache de Laravel pour suivre le nombre de requêtes par IP
17 * et par URL, bloquant temporairement les utilisateurs qui dépassent les limites définies.
18 *
19 * Fonctionnalités principales :
20 * - Limitation du nombre de requêtes par IP et par route.
21 * - Configuration flexible des seuils de déclenchement et de la durée de blocage.
22 * - Utilisation du cache Laravel pour un suivi efficace des tentatives.
23 * - Réinitialisation automatique des compteurs après expiration.
24 * - Réponse HTTP 429 (Too Many Requests) conforme aux standards.
25 * - Protection granulaire par route avec paramètres personnalisables.
26 *
27 * @package App\Http\Middleware
28 * @version 1.0
29 * @author Maxime ENTZ
30 * @since 1.0
31 * @note Middleware actuellement non utilisé selon le commentaire dans le code.
32 * @note Utilise le système de cache configuré dans l'application Laravel.
33 * @warning Peut bloquer des utilisateurs légitimes en cas de configuration trop restrictive.
34 * @see Cache Pour le stockage des compteurs de tentatives.
35 * @example
36 * ```php
37 * // Utilisation dans les routes (routes/web.php)
38 * Route::post('/contact', [ContactController::class, 'store'])
39 * ->middleware('antispam:10,5'); // 10 tentatives max en 5 minutes
40 *
41 * // Utilisation avec paramètres par défaut
42 * Route::post('/login', [AuthController::class, 'login'])
43 * ->middleware('antispam'); // 5 tentatives max en 1 minute
44 * ```
45 */

48 /**
49 * @brief Traite une requête entrante et applique la limitation de taux.
50 *
51 * Cette méthode constitue le cœur du middleware. Elle vérifie le nombre de tentatives
52 * effectuées par l'utilisateur depuis la même adresse IP vers la même URL, et bloque
53 * l'accès si le seuil autorisé est dépassé. Le système utilise une clé composite
54 * incluant l'URL complète et l'adresse IP pour un suivi granulaire.
55 *
56 * Le processus de vérification suit ces étapes :
57 * 1. Génération d'une clé unique basée sur l'URL et l'IP de l'utilisateur.
58 * 2. Incrémentation du compteur de tentatives dans le cache.
59 * 3. Configuration de l'expiration du compteur lors de la première tentative.
60 * 4. Vérification du dépassement du seuil autorisé.
61 * 5. Blocage ou autorisation de la requête selon le résultat.
62 *
63 * @param Request $request Requête HTTP entrante à analyser.
64 * @param Closure $next Fonction de callback pour passer au middleware suivant.
65 * @param int $maxAttempts Nombre maximum de tentatives autorisées (défaut: 5).
66 * @param int $decayMinutes Durée en minutes avant réinitialisation du compteur (défaut: 1).
67 *
68 * @return Response Réponse HTTP (429 si bloqué, sinon passage au middleware suivant).
69 *
70 * @throws \Illuminate\Contracts\Cache\LockTimeoutException Si le cache est indisponible.
71 *
72 * @example
73 * ```php
74 * // Configuration dans le kernel HTTP (app/Http/Kernel.php)
75 * protected $middlewareAliases = [
76 * 'antispam' => \App\Http\Middleware\AntiSpamMiddleware::class,
77 * ];
78 *
79 * // Utilisation restrictive pour les formulaires sensibles
80 * Route::post('/reset-password', function() {
81 * // Logic here
82 * })->middleware('antispam:3,10'); // 3 tentatives max en 10 minutes
83 * ```
84 *
85 * @example
86 * ```php
87 * // Utilisation pour protéger les API publiques
88 * Route::apiResource('users', UserController::class)
89 * ->middleware('antispam:50,60'); // 50 requêtes max par heure
90 * ```
91 *
92 * @note La clé de cache inclut l'URL complète pour permettre différents seuils par endpoint.
93 * @note Le premier appel à Cache::increment() retourne 1 et déclenche la configuration d'expiration.
94 * @note Les tentatives sont comptées par combinaison IP + URL, pas globalement par IP.
95 * @warning Une configuration trop restrictive peut nuire à l'expérience utilisateur.
96 * @warning Le middleware ne distingue pas les requêtes légitimes des tentatives malveillantes.
97 * @see Cache::increment() Pour l'incrémentation atomique des compteurs.
98 * @see now()->addMinutes() Pour la gestion de l'expiration des compteurs.
99 */
100 public function handle(Request $request, Closure $next, int $maxAttempts = 5, int $decayMinutes = 1): Response
101 {
102 // On utilise l'URL de la route et l'adresse IP de l'utilisateur.
103 $key = 'anti-spam:' . $request->fullUrl() . ':' . $request->ip();
104
105 // Cache::increment() retourne la nouvelle valeur après incrémentation.
106 $attempts = Cache::increment($key);
107
108 // Cela permet de réinitialiser le compteur après un certain temps.
109 if ($attempts === 1) {
110 Cache::put($key, 1, now()->addMinutes($decayMinutes));
111 }
112
113 if ($attempts > $maxAttempts) {
114 // return redirect()->back()->withErrors(['spam' => 'Trop de tentatives. Veuillez réessayer plus tard.']);
115 // ou
116 return response('Trop de tentatives de spam. Veuillez réessayer plus tard.', 429); // 429 Too Many Requests
117 }
118
119 // 5. Si pas de spam détecté, passer la requête au middleware suivant ou au contrôleur
120 return $next($request);
121 }
122}
