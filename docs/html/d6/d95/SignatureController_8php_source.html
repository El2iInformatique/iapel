99 /**
100 * @brief Traite la signature électronique et génère le document PDF certifié final.
101 *
102 * Cette méthode complexe gère tout le processus de signature électronique :
103 * - Validation de la signature base64 reçue du client.
104 * - Vérification et marquage du token comme utilisé.
105 * - Intégration de la signature et de la date dans le PDF à la position calculée.
106 * - Redimensionnement automatique de la signature en conservant les proportions.
107 * - Génération d'un PDF intermédiaire avec signature visuelle.
108 * - Certification électronique du document via script Python externe.
109 * - Nettoyage automatique des fichiers temporaires après traitement.
110 *
111 * Processus de positionnement :
112 * - Utilise un ratio de conversion précis (6.98) pour le placement.
113 * - Place automatiquement la date au format français (d/m/Y).
114 * - Redimensionne la signature dans un cadre 31x13 unités maximum.
115 * - Applique les modifications sur l'avant-dernière page du document.
116 *
117 * @param Request $request Requête POST contenant :
118 * - signature : Signature encodée en base64 (obligatoire)
119 * @param string $token Token unique d'identification du devis à signer.
120 *
121 * @return void Cette méthode ne retourne rien mais génère des fichiers sur le système.
122 *
123 * @throws Exception Si la validation de la signature échoue.
124 * @throws ProcessFailedException Si le script de certification Python échoue.
125 * @throws Exception Si une erreur survient lors de la manipulation des fichiers.
126 *
127 * @note Le ratio de conversion 6.98 est calibré spécifiquement pour les modèles PDF utilisés.
128 * @note La signature est redimensionnée automatiquement en conservant les proportions.
129 * @note Les fichiers temporaires (signature PNG, PDF signé) sont supprimés après traitement.
130 * @warning La certification électronique nécessite Python et les dépendances appropriées.
131 * @warning Les positions de signature doivent être préalablement configurées dans le token.
132 * @par Exemple:
133 * POST /signature/ABC123 avec signature base64 génère un PDF certifié.
134 */
135 public function sign(Request $request, $token)
136 {
137 $request->validate([
138 'signature' => 'required|string' // Signature encodée en base64
139 ]);
140
141 $tokenEntry = Token::where('token', $token)
142 //->where('used', false)
143 //->where('expires_at', '>', now())
144 ->first();
145
146 if (!$tokenEntry) {
147 //return response()->json(['message' => 'Token invalide ou expiré'], Response::HTTP_FORBIDDEN);
148 }
149
150 // Marquer le token comme utilisé
151 $tokenEntry->used = true;
152 $tokenEntry->save();
153
154 // On modifie le pdf pour inclure la signature du client
155 $uid = $tokenEntry->devis_id;
156 $document = 'devis';
157 $nomDoc = $uid . '_' . $token;
158 $client = $tokenEntry->organisation_id;
159 $pdfPath = storage_path('app/public/' . $client . '/' . $document . '/' . $nomDoc . '/' . $nomDoc . '.pdf'); // PDF du devis initial
160 $outputPath = storage_path('app/public/' . $client . '/' . $document . '/' . $nomDoc . '/' . $nomDoc . '_signe.pdf'); // PDF du devis avec signature client
161
162 // Charger le PDF source
163 $pdf = new Fpdi();
164 $pdf->SetAutoPageBreak(false);
165
166 // ombre total de pages dans le PDF source
167 $pageCount = $pdf->setSourceFile($pdfPath);
168
169 // Importer toutes les pages
170 for ($i = 1; $i <= $pageCount; $i++) {
171 $pdf->AddPage();
172 $tplIdx = $pdf->importPage($i);
173 $pdf->useTemplate($tplIdx, 0, 0, null, null, true);
174
175 // Si avant-dernière page, appliquer signature et date
176 if ($i == $pageCount - $tokenEntry->nb_pages) {
177 $hauteurSignature = $tokenEntry->position_signature;
178
179 $ratioConversion = 6.98; // ce 6.98 a l'air de sortir du chapeau mais je jure que c'est un ratio correct que j'ai calculé a la main
180
181 $xDate = $tokenEntry->x_date;
182 $yDate = $tokenEntry->y_date;
183
184 $xSignature = $tokenEntry->x_signature;
185 $ySignature = $tokenEntry->y_signature;
186
187 // Intégration de la date
188 $date_signature = date('d/m/Y');
189 $pdf->SetXY( $xDate/$ratioConversion, $yDate/$ratioConversion);
190 $pdf->Write(10, $date_signature);
191
192 // Intégration de la signature
193 $signatureBase64 = $request->input('signature');
194 $signatureData = base64_decode(preg_replace('#^data:image/\w+;base64,#i', '', $signatureBase64));
195 $signaturePath = storage_path('app/public/' . $client . '/' . $document . '/' . $nomDoc . '/' . $nomDoc . '_signature.png');
196 file_put_contents($signaturePath, $signatureData);
197 list($width, $height) = getimagesize($signaturePath); // Récupère la taille originale
198 $maxWidth = 31;
199 $maxHeight = 13;
200 // Calcul du redimensionnement en conservant le ratio
201 if ($width > $height) {
202 $newWidth = $maxWidth;
203 $newHeight = ($height / $width) * $maxWidth;
204 } else {
205 $newHeight = $maxHeight;
206 $newWidth = ($width / $height) * $maxHeight;
207 }
208 // Calcul des positions pour centrer dans le carré
209
210 $xImage = $xSignature / $ratioConversion;
211 $yImage = $ySignature / $ratioConversion;
212 $pdf->Image($signaturePath, $xImage, $yImage, $newWidth, $newHeight, '', '', 'T', false, 300, '', false, false, 0, false, false, false);
213
214 /** Mettre page de fin ici */
215
216 }
217 }
218
219 // Désactiver les en-têtes et pieds de page
220 $pdf->SetMargins(0, 0, 0);
221 $pdf->setPrintHeader(false);
222 $pdf->setPrintFooter(false);
223
224
225
226
227 // Aplatir le PDF en l'empêchant d'être modifié
228 $pdf->Output($outputPath, 'F');
229
230
231 // On va signer électroniquement le devis
232 $pdfSignePath = storage_path('app/public/' . $client . '/' . $document . '/' . $nomDoc . '/' . $nomDoc . '_signe.pdf'); // PDF du devis avec signature client
233 $pdfCertifiePath = storage_path('app/public/' . $client . '/' . $document . '/' . $nomDoc . '/' . $nomDoc . '_certifie.pdf'); // PDF du devis avec signature client
234 $scriptPath = storage_path('app/signature/sign.py');
235 $pythonPath = trim(shell_exec("which python3"));
236 $process = new Process([$pythonPath, $scriptPath, $pdfSignePath, $pdfCertifiePath]);
237 $process->run();
238
239 // 📌 Vérifier si le script a échoué
240 if (!$process->isSuccessful()) {
241 throw new ProcessFailedException($process); // Affiche l'erreur dans Laravel
242 }
243
244 // Suppression des fichiers temporaires
245 unlink($outputPath);
